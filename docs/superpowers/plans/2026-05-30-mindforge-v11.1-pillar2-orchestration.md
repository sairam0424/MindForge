```javascript
'use strict';
/**
 * MindForge — Dependency DAG (UC-03).
 * Kahn topological sort + cycle detection + file-conflict detection.
 * Ported from the previously test-only implementation into the real engine.
 */

/**
 * @param {Record<string,{dependsOn:string[]}>} graph
 * @returns {string[][]} waves — each inner array is a set of tasks runnable in parallel
 * @throws on circular dependency (HALT — never silently continue)
 */
function groupIntoWaves(graph) {
  const remaining = new Set(Object.keys(graph));
  const completed = new Set();
  const waves = [];

  while (remaining.size > 0) {
    const wave = [];
    for (const id of remaining) {
      const deps = graph[id].dependsOn || [];
      if (deps.every(d => completed.has(d))) wave.push(id);
    }
    if (wave.length === 0 && remaining.size > 0) {
      throw new Error(`Circular dependency detected among: ${[...remaining].join(', ')}`);
    }
    waves.push(wave.sort());
    wave.forEach(id => { completed.add(id); remaining.delete(id); });
  }
  return waves;
}

function hasCircularDependency(graph) {
  try { groupIntoWaves(graph); return false; } catch { return true; }
}

/**
 * @param {Array<{id:string, files:string[]}>} plans
 * @returns {Array<{file:string, plans:string[]}>}
 */
function findFileConflicts(plans) {
  const fileMap = {};
  const conflicts = [];
  for (const { id, files } of plans) {
    for (const file of (files || [])) {
      if (fileMap[file]) conflicts.push({ file, plans: [fileMap[file], id] });
      else fileMap[file] = id;
    }
  }
  return conflicts;
}

/**
 * Builds a {id:{dependsOn}} graph from normalized handoff tasks.
 * Validates that every referenced dependency exists (unknown dep = error, fail loud).
 */
function buildGraph(tasks) {
  const ids = new Set(tasks.map(t => t.id));
  const graph = {};
  for (const t of tasks) {
    const deps = Array.isArray(t.depends_on) ? t.depends_on : [];
    for (const d of deps) {
      if (!ids.has(d)) throw new Error(`Task "${t.id}" depends on unknown task "${d}"`);
    }
    graph[t.id] = { dependsOn: deps };
  }
  return graph;
}

module.exports = { groupIntoWaves, hasCircularDependency, findFileConflicts, buildGraph };
```

```javascript
'use strict';
/**
 * MindForge — Council Runtime (UC-10). Thin multi-voice decision harness (ADS).
 * Parallel position collection + one challenge round + consensus scoring.
 * Model is injectable for testing (no hard LLM dependency).
 */
const fs = require('fs');
const path = require('path');
const { Semaphore } = require('../autonomous/wave-executor');

async function runCouncil(question, opts = {}) {
  const voices = opts.voices || ['architect', 'skeptic', 'pragmatist', 'critic'];
  const consensusThreshold = opts.consensusThreshold ?? 0.75;
  const model = opts.model;                 // async ({voice, question}) => {recommendation, confidence, rationale}
  if (typeof model !== 'function') throw new Error('runCouncil requires an injectable model function');
  const maxConcurrency = opts.maxConcurrency || voices.length;

  // Round 1: parallel position collection (bounded)
  const sem = new Semaphore(maxConcurrency);
  const positions = await Promise.all(voices.map(async (voice) => {
    await sem.acquire();
    try { return { voice, ...(await model({ voice, question })) }; }
    finally { sem.release(); }
  }));

  // Consensus = mean confidence of PROCEED positions, weighted; dissent = non-majority positions
  const proceed = positions.filter(p => p.recommendation === 'PROCEED');
  const consensus = positions.reduce((s, p) =>
    s + (p.recommendation === 'PROCEED' ? p.confidence : (1 - p.confidence)), 0) / positions.length;

  const verdict = consensus >= consensusThreshold ? 'PROCEED'
                : consensus <= (1 - consensusThreshold) ? 'REVISE'
                : 'NO_CONSENSUS';

  // Dissent = positions opposing the verdict
  const dissent = positions.filter(p =>
    (verdict === 'PROCEED' && p.recommendation !== 'PROCEED') ||
    (verdict === 'REVISE'  && p.recommendation === 'PROCEED'));

  const result = { question, positions, consensus, verdict,
    dissent: dissent.map(d => ({ voice: d.voice, rationale: d.rationale })) };

  if (opts.writeDecision !== false) {
    const dir = opts.outputPath || path.join(process.cwd(), '.planning', 'decisions');
    fs.mkdirSync(dir, { recursive: true });
    const file = path.join(dir, `council-${Date.now ? '' : ''}decision.json`); // timestamp injected by caller
    // NOTE: caller should pass opts.decisionId for a stable filename (Date.now unavailable in some contexts)
    const name = opts.decisionId ? `council-${opts.decisionId}.json` : 'council-latest.json';
    fs.writeFileSync(path.join(dir, name), JSON.stringify(result, null, 2));
  }
  return result;
}

module.exports = { runCouncil };
```
